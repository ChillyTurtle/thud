//Health Bonus Pulse
event HudHealthBonusPulse
{
	
	Animate PlayerStatusHealthValueShadow	FgColor		"Health Buff Solid"		Linear 0.0 0.75
	Animate PlayerStatusHealthValueShadow	FgColor		"Health Buff Pulse"	Linear 0.85 0.75
	Animate PlayerStatusHealthValueShadow2	Alpha		"255"				Linear 0.85 0.75
	Animate	PlayerStatusHealthBonusImage	Alpha		"255"				Linear 0.0 0.0
	
	RunEvent HudHealthBonusPulseLoop	1.70
}

// call to loop HudHealthBonusPulse
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
	
	Animate PlayerStatusHealthValueShadow		FgColor	"Health Value Shadow"		Linear 0.01 0.25
	Animate PlayerStatusHealthValueShadow2		Alpha	"0"				Linear 0.01 0.25
	Animate	PlayerStatusHealthBonusImage		Alpha	"0"				Linear 0.0 0.0
}

// Spy Disguise
event HudSpyDisguiseChanged
{
	Animate PlayerStatusSpyOutlineImage		Alpha		"255"			Linear 0.0 0.2
	
	Animate PlayerStatusSpyOutlineImage		Position	"c-250 c-50"	Linear 0.0 0.2		// 200 200
	Animate PlayerStatusSpyOutlineImage		Size		"150 150"		Linear 0.0 0.2		// 400 400

	RunEvent HudSpyDisguiseHide	0.7
}

event HudSpyDisguiseHide
{
	Animate PlayerStatusSpyOutlineImage		Position	"c-290 r90"		Linear 0.0 0.2
	Animate PlayerStatusSpyOutlineImage		Size		"40 40"			Linear 0.0 0.2
	
	Animate PlayerStatusSpyOutlineImage		Alpha		"0"				Linear 0.2 0.1
}

	Animate PlayerStatusSpyImage			Alpha		"0"				Linear 0.9 0.1	
	//Animate PlayerStatusClassImage		Size		"0 0"			Linear 0.0 0.0
}
